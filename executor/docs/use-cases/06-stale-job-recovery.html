switch (buildStatus.phase()) {
    case BUILDING -> {
        dispatchService.markAsRunning(job, job.getBuildNo());
        jobPort.save(job);
        notifyStartedPort.notify(...);
    }
    case COMPLETED -> {
        dispatchService.markAsCompleted(job, buildStatus.result());
        jobPort.save(job);
        notifyCompletedPort.notify(...);
    }
